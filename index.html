document.getElementById('acessar-arquivo').onclick = async function() { try { const token = await magic.user.getIdToken(); const fileName = "Paper.pdf"; const response = await fetch("https://fleek-nine.vercel.app/api/get-file", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ token, fileName }) }); if (!response.ok) { throw new Error(`Erro ao acessar o backend: ${response.statusText}`); } const pdfBlob = await response.blob(); const pdfUrl = URL.createObjectURL(pdfBlob); // Abre o PDF em uma nova aba window.open(pdfUrl, "_blank"); } catch (err) { console.error("Erro na requisição:", err.message); alert("Ocorreu um erro: " + err.message); } };
